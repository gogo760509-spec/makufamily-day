<script>
    // 載入 YouTube Iframe Player API
    var tag = document.createElement('script');
    tag.src = "https://www.youtube.com/iframe_api";
    var firstScriptTag = document.getElementsByTagName('script')[0];
    firstScriptTag.parentNode.insertBefore(tag, firstScriptTag);

    var player;
    var currentButton = null;
    var stopTimeout;
    const PLAY_DURATION = 30000; // 試聽時間：30秒 (30000毫秒)

    // API 載入完成後的初始化
    function onYouTubeIframeAPIReady() {
        player = new YT.Player('yt-player', {
            height: '0',
            width: '0',
            videoId: '',
            playerVars: {
                'playsinline': 1,
                'controls': 0,
                'disablekb': 1
            },
            events: {
                'onStateChange': onPlayerStateChange
            }
        });
    }

    // 控制音樂播放與暫停
    function togglePlay(videoId, startTime, button) {
        if (!player || typeof player.cueVideoById !== 'function') {
            alert('音樂播放器準備中，請稍候再試！');
            return;
        }

        // 如果點擊的是正在播放的歌曲，則暫停
        if (currentButton === button && player.getPlayerState() === YT.PlayerState.PLAYING) {
            player.pauseVideo();
            resetButton(button);
            return;
        }

        // 清除先前的定時器與按鈕狀態
        if (currentButton) {
            resetButton(currentButton);
        }
        clearTimeout(stopTimeout);

        // 更新當前操作的按鈕
        currentButton = button;
        button.innerText = "■ 停止";
        button.classList.add('playing');

        // 載入並播放指定影片及起始秒數
        player.loadVideoById({
            videoId: videoId,
            startSeconds: startTime
        });

        // 設定 30 秒後自動停止
        stopTimeout = setTimeout(function() {
            if (player && player.getPlayerState() === YT.PlayerState.PLAYING) {
                player.pauseVideo();
                resetButton(button);
            }
        }, PLAY_DURATION);
    }

    function resetButton(button) {
        button.innerText = "▶ 試聽";
        button.classList.remove('playing');
    }

    function onPlayerStateChange(event) {
        // 如果影片播完了或被手動關閉，重設按鈕狀態
        if (event.data === YT.PlayerState.ENDED || event.data === YT.PlayerState.PAUSED) {
            if (currentButton && event.data === YT.PlayerState.ENDED) {
                resetButton(currentButton);
                clearTimeout(stopTimeout);
            }
        }
    }

    // 表單送出邏輯
    function submitForm(event) {
        event.preventDefault();
        
        // 停止目前正在播放的音樂
        if (player && player.getPlayerState() === YT.PlayerState.PLAYING) {
            player.pauseVideo();
            if (currentButton) resetButton(currentButton);
        }

        const nickname = document.getElementById('nickname').value;
        const selectedSong = document.querySelector('input[name="selectedSong"]:checked').value;
        const message = document.getElementById('message').value;

        // 顯示成功提示（後續可在此處串接 GAS API 寫入 Google 試算表）
        alert(`點歌成功！\n\n同仁暱稱：${nickname}\n點選歌曲：${selectedSong}\n留言內容：${message}`);
        
        // 重設表單
        document.getElementById('voteForm').reset();
    }
</script>
